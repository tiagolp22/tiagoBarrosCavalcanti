Variables 
    tableau lesEntiers[3] : entier 
    somme, i : entier 
    moyenne : reel 
Début 
    //Boucle de remplissage 
    Pour i allant de 0 à 2 
        Ecrire("Entrez un nombre.")
        Lire(lesEntiers[i])
    Fin Pour 

    somme <- 0 
    Pour i allant de 0 à 2 
        somme <- somme + lesEntiers[i]
    Fin Pour 

    moyenne <- somme / 3
Fin

Variables 
    tableau lesEntiers[15] : entier 
    somme, i : entier 
    moyenne : reel 
Début 
    //Boucle de remplissage 
    Pour i allant de 0 à Taille(lesEntiers) - 1
        Ecrire("Entrez un nombre.")
        Lire(lesEntiers[i])
    Fin Pour 

    somme <- 0 
    Pour i allant de 0 à Taille(lesEntiers) - 1
        somme <- somme + lesEntiers[i]
    Fin Pour 

    moyenne <- somme / Taille(lesEntiers)
Fin
